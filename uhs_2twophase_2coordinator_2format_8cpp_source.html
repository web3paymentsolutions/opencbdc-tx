27 auto operator<<(serializer& ser,
28 const coordinator::controller::sm_command& c)
29 -> serializer& {
30 ser << c.m_header;
31 switch(c.m_header.m_comm) {
32 case coordinator::state_machine::command::prepare: {
33 const auto& data
34 = std::get<coordinator::controller::prepare_tx>(
35 c.m_data.value());
36 ser << data;
37 break;
38 }
39 case coordinator::state_machine::command::commit: {
40 const auto& data
41 = std::get<coordinator::controller::commit_tx>(
42 c.m_data.value());
43 ser << data;
44 break;
45 }
46 // Discard, done and get don't have a payload
47 case coordinator::state_machine::command::discard:
48 case coordinator::state_machine::command::done:
49 [[fallthrough]];
50 case coordinator::state_machine::command::get: {
51 break;
52 }
53 }
54 return ser;
55 }
