36 auto node::init(const nuraft::raft_params& raft_params) -> bool {
37 // Set these explicitly as they depend on the structure of this class
38 auto params = raft_params;
39 if(m_blocking) {
40 params.return_method_ = nuraft::raft_params::blocking;
41 } else {
42 params.return_method_ = nuraft::raft_params::async_handler;
43 }
44 params.auto_forwarding_ = false;
45
46 m_raft_instance = m_launcher.init(m_sm,
47 m_smgr,
48 m_raft_logger,
49 m_port,
50 m_asio_opt,
51 params,
52 m_init_opts);
53
54 if(!m_raft_instance) {
55 m_log->error("Failed to initialize raft launcher");
56 return false;
57 }
58
59 m_log->info("Waiting for raft initialization");
60 static constexpr auto wait_time = std::chrono::milliseconds(100);
61 while(!m_raft_instance->is_initialized()) {
62 std::this_thread::sleep_for(wait_time);
63 }
64 m_log->info("Raft initialization complete");
65
66 return true;
67 }

73 auto node::replicate(nuraft::ptr<nuraft::buffer> new_log,
74 const callback_type& result_fn) const -> bool {
75 auto ret = m_raft_instance->append_entries({std::move(new_log)});
76 if(!ret->get_accepted()) {
77 return false;
78 }
79
80 if(result_fn) {
81 ret->when_ready(result_fn);
82 }
83
84 return true;
85 }
