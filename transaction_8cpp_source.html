122 compact_tx::compact_tx(const full_tx& tx) {
123 m_id = tx_id(tx);
124 for(const auto& inp : tx.m_inputs) {
125 m_inputs.push_back(inp.hash());
126 }
127 for(uint64_t i = 0; i < tx.m_outputs.size(); i++) {
128 m_uhs_outputs.push_back(
129 uhs_id_from_output(m_id, i, tx.m_outputs[i]));
130 }

133 auto compact_tx::sign(secp256k1_context* ctx, const privkey_t& key) const
134 -> sentinel_attestation {
135 auto payload = hash();
136 auto pubkey = pubkey_from_privkey(key, ctx);
137 secp256k1_keypair keypair{};
138 [[maybe_unused]] const auto ret
139 = secp256k1_keypair_create(ctx, &keypair, key.data());
140 assert(ret == 1);
141
142 auto sig = signature_t();
143 [[maybe_unused]] const auto sign_ret
144 = secp256k1_schnorrsig_sign(ctx,
145 sig.data(),
146 payload.data(),
147 &keypair,
148 nullptr,
149 nullptr);
150 assert(sign_ret == 1);
151 return {pubkey, sig};
152 }

166 auto compact_tx::verify(secp256k1_context* ctx,
167 const sentinel_attestation& att) const -> bool {
168 auto payload = hash();
169 secp256k1_xonly_pubkey pubkey{};
170 if(secp256k1_xonly_pubkey_parse(ctx, &pubkey, att.first.data()) != 1) {
171 return false;
172 }
173
174 if(secp256k1_schnorrsig_verify(ctx,
175 att.second.data(),
176 payload.data(),
177 &pubkey)
178 != 1) {
179 return false;
180 }
181
182 return true;
183 }
